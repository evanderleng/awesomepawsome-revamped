pipeline {
    agent any
    stages {
        stage('Checkout SCM') {
            steps {
                script {
                    env.GIT_CREDENTIALS = credentials('GIT_CREDENTIALS_ID')
                }
                git branch: 'main', url: 'https://github.com/fwishyy/awesome-pawsome.git', credentialsId: env.GIT_CREDENTIALS
            }
        }

        stage('OWASP DependencyCheck') {
            steps {
                dependencyCheck additionalArguments: '--format HTML --format XML', odcInstallation: 'Default'
            }
        }
    }
    post {
        success {
            dependencyCheckPublisher pattern: 'dependency-check-report.xml'
        }
    }
}
